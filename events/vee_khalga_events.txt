# Vee begins to grow
narrative_event = {
    id = KHAVEE.1704
    title = EVTTITLE_KHAVEE.1704
    desc = EVTDESC_KHAVEE.1704
    picture = GFX_evt_whispers
    border = GFX_event_narrative_frame_war

    is_triggered_only = yes
    
    has_character_flag = is_vee
    capable_only = yes  
    
    trigger = {
        age = 13
        OR = {
            OR = {
                ai = no             
            }
            father = { ai = no }
            mother = { ai = no }
        }
        NOT = { has_character_flag = vee_spawn_rumors }
    }
    
    option = {
        name = EVTOPTA_KHAVEE.1704
        set_character_flag = vee_spawn_rumors
        hidden_tooltip = {
            father = { narrative_event = { id = KHAVEE.1705 } }
            mother = { narrative_event = { id = KHAVEE.1705 } }
        }
        prestige = 20
        if = {
            limit = { NOT = { trait = proud } }
            random = {
                chance = 30
                add_trait = proud
                hidden_tooltip = {
                    character_event = {
                        id = 38261 #Notify Proud
                    }
                }
            }
        }
    }
}

### Select a friend to get The Mangekyo ###
character_event = {
    id = KHAVEE.1706
    desc = EVTDESC100000
    picture = GFX_evt_child_sword

    is_triggered_only = yes
    
    min_age = 4
    max_age = 14
    only_capable = yes
    prisoner = no
    
    trigger = {
        trait = sharingan
        NOT = {
            num_of_friends = 2
        }
        NOT = {
            num_of_rivals = 2
        }       

        host = {
            any_courtier = {
                is_abroad = no
                prisoner = no
                at_location = ROOT

                age = 4
                NOT = {
                    age = 14
                }
                NOT = {
                    num_of_friends = 2
                }
                NOT = {
                    num_of_rivals = 2
                }
                NOT = {
                    is_friend = ROOT
                }
                NOT = {
                    is_rival = ROOT
                }
            }
        }
    }

    immediate = {
        host = {
            random_courtier = {
                limit = {
                    is_abroad = no
                    prisoner = no
                    at_location = ROOT                  
                    
                    age = 4
                    NOT = {
                        age = 14
                    }
                    NOT = {
                        num_of_friends = 2
                    }
                    NOT = {
                        num_of_rivals = 2
                    }
                    NOT = {
                        is_friend = ROOT
                    }
                    NOT = {
                        is_rival = ROOT
                    }
                }
                save_event_target_as = troublemaker_child
            }
        }
    }   
    
    option = {
        name = EVTOPTB100000
        event_target:troublemaker_child = {
            add_friend = ROOT
            character_event = {
                id = 100002
                tooltip = EVTTOOLTIP100002
            }
        }
    }
}
